    <script>
        document.addEventListener("DOMContentLoaded", function () {
            const loadMoreBtn = document.getElementById("load-more");
            const postColumns = document.querySelectorAll(".blog-section__col");
            if (!loadMoreBtn) return;

            loadMoreBtn.addEventListener("click", function () {
                const cursorDate = loadMoreBtn.getAttribute("data-cursor-date") || "";
                const cursorId   = loadMoreBtn.getAttribute("data-cursor-id") || "";

                const formData = new FormData();
                formData.append("action", "load_more");
                if (cursorDate) formData.append("cursor_date", cursorDate);
                if (cursorId)   formData.append("cursor_id", cursorId);

                fetch("https://betatransfer.io/wp-admin/admin-ajax.php", { method: "POST", body: formData })
                    .then(r => r.text())
                    .then(html => {
                        if (html.trim() === "end") {
                            loadMoreBtn.style.display = "none";
                            return;
                        }

                        const temp = document.createElement("div");
                        temp.innerHTML = html;
                        const cards = temp.querySelectorAll(".blog-card");
                        if (!cards.length) { loadMoreBtn.style.display = "none"; return; }

                        cards.forEach((card) => {
                            const leftCount  = postColumns[0].querySelectorAll(".blog-card").length;
                            const rightCount = postColumns[1].querySelectorAll(".blog-card").length;
                            (leftCount <= rightCount ? postColumns[0] : postColumns[1]).appendChild(card);
                        });

                        const last = cards[cards.length - 1];
                        loadMoreBtn.setAttribute("data-cursor-date", last.getAttribute("data-post-date"));
                        loadMoreBtn.setAttribute("data-cursor-id",   last.getAttribute("data-post-id"));

                        document.dispatchEvent(new Event("ajaxContentLoaded"));
                    })
                    .catch(err => console.error("error load posts:", err));
            });
